```bash
mkdir -p "$RUN_DIR/raw"
GAAD_FILE="$RUN_DIR/raw/iam_gaad.json"
aws iam get-account-authorization-details \
  --filter User Role Group LocalManagedPolicy \
  --output json > "$GAAD_FILE" 2>&1
```

Check for AccessDenied:
```bash
if grep -q "AccessDenied\|UnauthorizedAccess" "$GAAD_FILE"; then
  echo "[WARN] GAAD AccessDenied, falling back to per-resource enumeration"
  # ... fallback path (see below)
else
  GAAD=$(cat "$GAAD_FILE")
  # ... GAAD primary path
fi
```

Do NOT add `--no-paginate` — AWS CLI v2 auto-paginates by default. Adding it would silently truncate results (PAGE-01, PAGE-02).

**2. AssumeRolePolicyDocument handling (IAM-03):**


**3. GAAD path extraction templates (stdin piping per PERF-01):**

Write each resource type's findings to a temp file, then combine:

**iam_user from GAAD:**
```bash
echo "$GAAD" | jq --arg account_id "$ACCOUNT_ID" '[.UserDetailList[] | {
  resource_type: "iam_user",
  resource_id: .UserName,
  arn: .Arn,
  region: "global",
  created: .CreateDate,
  groups: [.GroupList[]],
  attached_policies: [.AttachedManagedPolicies[] | .PolicyArn],
  inline_policies: (.UserPolicyList // [] | map({name: .PolicyName, document: .PolicyDocument})),
  has_mfa: false,
  has_console_access: false,
  access_keys: [],
  findings: []
}]' > "$RUN_DIR/raw/iam_users_base.json"
```
Note: has_mfa, has_console_access, access_keys are placeholders — filled by per-user credential-state loop below.

**iam_role from GAAD (uses TRUST_CLASSIFY_JQ):**
```bash
echo "$GAAD" | jq --arg account_id "$ACCOUNT_ID" \
  "$TRUST_CLASSIFY_JQ"'
  [.RoleDetailList[] |
    select(.RoleName | startswith("AWSServiceRole") | not) | . as $role |
    ($role.AssumeRolePolicyDocument.Statement // []) as $stmts |
    ... (trust classification logic from existing agent) ...
    {
      resource_type: "iam_role",
      resource_id: $role.RoleName,
      arn: $role.Arn,
      region: "global",
      trust_relationships: $trust,
      is_service_linked: ($role.Path | startswith("/aws-service-role/")),
      permission_boundary: ($role.PermissionsBoundary.PermissionsBoundaryArn // null),
      role_last_used: ($role.RoleLastUsed | if . then {last_used_date: .LastUsedDate, region: .Region} else null end),
      inline_policies: ($role.RolePolicyList // [] | map({name: .PolicyName, document: .PolicyDocument})),
      findings: []
    }
  ]' > "$RUN_DIR/raw/iam_roles.json"
```
